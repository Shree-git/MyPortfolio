        def merge_sort(arr):
          if len(arr)>1:
              m = len(arr)//2
              L = arr[:m]
              R = arr[m:]
      
              merge_sort(L)
              merge_sort(R)
      
              i=j=k=0
              while i < len(L) and j < len(R):

                      i+=1

                      arr[k]=R[j]
                      j+=1
                  k+=1
